    const initialData = {
        riskClients: [
            { id: 1, name: 'Cliente Exemplo A', scores: [0,1,0,1,0,1], coordenador: 'João Silva', analista: 'Maria Santos', rtv: 'Pedro Costa', bordos: 100, clima: 50, hectare: 250, notes: 'Tudo OK' },

            { id: 3, name: 'Cliente Exemplo C', scores: [5,5,4,5,5,5], coordenador: 'Roberto Lima', analista: 'Fernanda Gomes', rtv: 'Marcelo Alves', bordos: 150, clima: 80, hectare: 350, notes: 'Alto risco' }
        ],
        actions: [
            { id: 1, area: 'Mapear Contratos', what: 'Verificar todos os contratos com término em 2026', who: 'Gerente Regional', start: '2026-03-01', end: '2026-03-15', status: 'Em Andamento' }

    let actionPlanChart, operationalChart;

    function getFullData() { const data = {}; for (const key in initialData) { const saved = localStorage.getItem(`dashboardData_${key}`); data[key] = saved ? JSON.parse(saved) : (initialData[key] || []); } return data; }
    function getData(key) { return getFullData()[key]; }
    function setData(key, data) { localStorage.setItem(`dashboardData_${key}`, JSON.stringify(data)); }
    function logChange(action) { const userName = document.getElementById('userName').value.trim() || 'Anônimo'; const log = getData('log'); const timestamp = new Date().toLocaleString('pt-BR'); log.unshift({ timestamp, user: userName, action }); setData('log', log); renderLog(); }

    function renderAll() { renderRiskMatrix(); renderActionTable(); renderGimbTable(); renderLog(); updateDashboard(); }
    function calculateRisk(scores) { const total = scores.reduce((sum, score) => sum + (parseInt(score, 10) || 0), 0); if (total <= 10) return { total, classification: '🟢 Seguro', className: 'risk-safe' }; if (total <= 18) return { total, classification: '🟡 Atenção', className: 'risk-warn' }; return { total, classification: '🔴 Alto Risco', className: 'risk-high' }; }
    
    function renderRiskMatrix() {
        const tableBody = document.querySelector('#riskMatrixTable tbody');
        tableBody.innerHTML = '';
        getData('riskClients').forEach(client => {
            const row = tableBody.insertRow();
            row.dataset.id = client.id;
            const risk = calculateRisk(client.scores);
            let scoreInputs = client.scores.map((score, index) => `<td><input type="number" min="0" max="5" class="risk-score-input" data-index="${index}" value="${score || 0}"></td>`).join('');
            row.innerHTML = `<td contenteditable="true" data-key="name">${client.name}</td>${scoreInputs}<td class="score-cell">${risk.total}</td><td class="classification-cell ${risk.className}">${risk.classification}</td><td contenteditable="true" data-key="coordenador">${client.coordenador}</td><td contenteditable="true" data-key="analista">${client.analista}</td><td contenteditable="true" data-key="rtv">${client.rtv}</td><td><input type="number" class="risk-score-input" data-key="bordos" value="${client.bordos || 0}" min="0"></td><td><input type="number" class="risk-score-input" data-key="clima" value="${client.clima || 0}" min="0"></td><td><input type="number" class="risk-score-input" data-key="hectare" value="${client.hectare || 0}" min="0"></td><td contenteditable="true" data-key="notes">${client.notes}</td>`;
        });
        updateTopRisks();
    }

    function renderActionTable() {
        const tableBody = document.querySelector('#actionPlanTable tbody');
        tableBody.innerHTML = '';
        getData('actions').forEach(action => {
            const row = tableBody.insertRow();
            row.dataset.id = action.id;
            row.innerHTML = `<td contenteditable="true" data-key="area">${action.area}</td><td contenteditable="true" data-key="what">${action.what}</td><td contenteditable="true" data-key="who">${action.who}</td><td contenteditable="true" data-key="start">${action.start}</td><td contenteditable="true" data-key="end">${action.end}</td><td contenteditable="true" data-key="status">${action.status}</td>`;
        });
        updateActionPlanChart();
    }

    function renderGimbTable() {
        const tableBody = document.querySelector('#gimbTable tbody');
        tableBody.innerHTML = '';
        getData('gimb').forEach(item => {
            const row = tableBody.insertRow();
            row.dataset.id = item.id;
            row.innerHTML = `<td contenteditable="true" data-key="item">${item.item}</td><td contenteditable="true" data-key="status">${item.status}</td><td contenteditable="true" data-key="notes">${item.notes}</td>`;
        });
    }

    function renderLog() { const logContainer = document.getElementById('logContainer'); logContainer.innerHTML = ''; getData('log').forEach(entry => { logContainer.innerHTML += `<div class="log-entry"><span class="timestamp">[${entry.timestamp}]</span> <span class="user">${entry.user}</span>: ${entry.action}</span></div>`; }); }
    
    function updateTopRisks() { const list = document.getElementById('topRisksList'); list.innerHTML = ''; const clients = getData('riskClients').map(c => ({ name: c.name, ...calculateRisk(c.scores) })).sort((a, b) => b.total - a.total); const top5 = clients.slice(0, 5); if (top5.length === 0 || top5[0].total === 0) { list.innerHTML = '<li>Nenhum risco significativo.</li>'; return; } top5.forEach(client => { if (client.total > 0) { const li = document.createElement('li'); li.className = client.className; li.innerHTML = `<span>${client.name}</span> <span class="score">${client.total}</span>`; list.appendChild(li); } }); }
    
    function updateDashboard() {
        const clients = getData('riskClients');
        const totalBordos = clients.reduce((sum, c) => sum + (parseInt(c.bordos) || 0), 0);
        const totalClima = clients.reduce((sum, c) => sum + (parseInt(c.clima) || 0), 0);
        const totalHectare = clients.reduce((sum, c) => sum + (parseInt(c.hectare) || 0), 0);
        const clientesRisco = clients.filter(c => calculateRisk(c.scores).total >= 19).length;
        const mediaRisco = clients.length > 0 ? Math.round(clients.reduce((sum, c) => sum + calculateRisk(c.scores).total, 0) / clients.length) : 0;

        document.getElementById('totalBordos').textContent = totalBordos.toLocaleString('pt-BR');
        document.getElementById('totalClima').textContent = totalClima.toLocaleString('pt-BR');
        document.getElementById('totalHectare').textContent = totalHectare.toLocaleString('pt-BR');
        document.getElementById('totalClientes').textContent = clients.length;
        document.getElementById('clientesRisco').textContent = clientesRisco;
        document.getElementById('mediaRisco').textContent = mediaRisco;

        updateOperationalChart(clients);
    }

    function updateOperationalChart(clients) {
        const data = {
            labels: clients.map(c => c.name),
            datasets: [
                { label: 'Bordos', data: clients.map(c => c.bordos || 0), backgroundColor: 'rgba(0, 123, 255, 0.7)' },
                { label: 'Clima', data: clients.map(c => c.clima || 0), backgroundColor: 'rgba(40, 167, 69, 0.7)' },
                { label: 'Hectare', data: clients.map(c => c.hectare || 0), backgroundColor: 'rgba(255, 193, 7, 0.7)' }
            ]
        };
        const ctx = document.getElementById('operationalChart').getContext('2d');
        if (operationalChart) { operationalChart.data = data; operationalChart.update(); } 
        else { operationalChart = new Chart(ctx, { type: 'bar', data: data, options: { responsive: true, scales: { y: { beginAtZero: true } } } }); }
    }

    function updateActionPlanChart() {
        const actions = getData('actions');
        const today = dayjs();
        const counts = { onTime: 0, dueSoon: 0, overdue: 0 };
        actions.forEach(action => {
            if (!action.end || action.status === 'Concluído') return;
            const endDate = dayjs(action.end);
            if (endDate.isBefore(today, 'day')) {
                counts.overdue++;
            } else if (endDate.diff(today, 'day') <= 7) {
                counts.dueSoon++;
            } else {
                counts.onTime++;
            }
        });
        const data = {
            labels: ['No Prazo', 'Próximo do Vencimento', 'Fora do Prazo'],
            datasets: [{ label: 'Status do Plano de Ação', data: [counts.onTime, counts.dueSoon, counts.overdue], backgroundColor: ['rgba(40, 167, 69, 0.7)', 'rgba(255, 193, 7, 0.7)', 'rgba(220, 53, 69, 0.7)'] }]
        };
        const ctx = document.getElementById('actionPlanChart').getContext('2d');
        if (actionPlanChart) { actionPlanChart.data = data; actionPlanChart.update(); } 
        else { actionPlanChart = new Chart(ctx, { type: 'bar', data: data, options: { indexAxis: 'y', responsive: true, plugins: { legend: { display: false } } } }); }
    }

    function addRiskClientRow() { const clients = getData('riskClients'); const newId = (clients.length > 0 ?     function addActionRow() { const actions = getData('actions'); const newId = (actions.length > 0 ?     function addGimbRow() { const gimb = getData('gimb'); const newId = (gimb.length > 0 ? Math.max(...gimb.map(g => g.id)) : 0) + 1; const newItem = { id: newId, item: 'Novo Item do Checklist', status: 'Pendente', notes: '' }; gimb.push(newItem); setData('gimb', gimb); logChange(`adicionou item ao GIMB: "${newItem.item}"`); renderGimbTable(); }

    function setupTableListener(tableId, dataKey, identifierKey) { 
        document.getElementById(tableId).addEventListener('focusout', (e) => { 
            if (e.target.isContentEditable && e.target.dataset.key) { 
                const row = e.target.closest('tr'); 
                const id = parseInt(row.dataset.id); 
                const key = e.target.dataset.key; 
                const newValue = e.target.innerText; 
                const data = getData(dataKey); 
                const item = data.find(i => i.id === id); 
                if (item && item[key] !== newValue) { 
                    const oldValue = item[key]; 
                    item[key] = newValue; 
                    setData(dataKey, data); 
                    logChange(`alterou "${key}" de "${oldValue}" para "${newValue}" em "${item[identifierKey]}"`); 
                    if (dataKey === 'riskClients') { renderRiskMatrix(); updateDashboard(); }
                    if (dataKey === 'actions') renderActionTable(); 
                } 
            } 
        }); 
    }

    document.addEventListener('DOMContentLoaded', () => { 
        renderAll(); 
        document.getElementById('userName').value = localStorage.getItem('dashboardUserName') || ''; 
        setupTableListener('riskMatrixTable', 'riskClients', 'name'); 
        setupTableListener('actionPlanTable', 'actions', 'area'); 
        setupTableListener('gimbTable', 'gimb', 'item'); 
    });
    
    document.getElementById('userName').addEventListener('change', (e) => { localStorage.setItem('dashboardUserName', e.target.value); });
    
    document.getElementById('riskMatrixTable').addEventListener('change', (e) => { 
        if (e.target.classList.contains('risk-score-input')) { 
            const row = e.target.closest('tr'); 
            const id = parseInt(row.dataset.id); 
            const clients = getData('riskClients'); 
            const client = clients.find(c => c.id === id); 
            if (e.target.dataset.index !== undefined) {
                client.scores[e.target.dataset.index] = parseInt(e.target.value, 10) || 0;
            } else if (e.target.dataset.key) {
                client[e.target.dataset.key] = parseInt(e.target.value, 10) || 0;
            }
            setData('riskClients', clients); 
            logChange(`atualizou scores do cliente "${client.name}"`); 
            renderRiskMatrix(); 
            updateDashboard();
        } 
    });
    
    function exportToPDF() { const { jsPDF } = window.jspdf; const activeTab = document.querySelector('.tab-content.active'); if (!activeTab) { alert("Nenhuma aba ativa para exportar."); return; } logChange(`iniciou exportação para PDF da aba "${activeTab.id}".`); html2canvas(activeTab, { scale: 2, useCORS: true, backgroundColor: '#ffffff' }).then(canvas => { const imgData = canvas.toDataURL('image/png'); const pdf = new jsPDF({ orientation: 'landscape', unit: 'px', format: [canvas.width, canvas.height] }); pdf.addImage(imgData, 'PNG', 0, 0, canvas.width, canvas.height); pdf.save('dashboard_relatorio.pdf'); }).catch(err => { console.error("Erro ao gerar o PDF:", err); alert("Ocorreu um erro ao gerar o PDF."); }); }
